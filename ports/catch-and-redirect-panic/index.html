	file, err := os.OpenFile(path, os.O_RDWR|os.O_CREATE|os.O_APPEND, 0666)
	globalf = file
	if err != nil {
		return err

	file.Write([]byte(t))
	file.Write([]byte("\n"))

	if err = syscall.Dup2(int(file.Fd()), int(os.Stderr.Fd())); err != nil {
		return err
	}
	return nil
}
